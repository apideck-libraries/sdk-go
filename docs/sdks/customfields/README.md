# CustomFields
(*Vault.CustomFields*)

## Overview

### Available Operations

* [List](#list) - Get resource custom fields

## List

This endpoint returns an custom fields on a connection resource.


### Example Usage

```go

import(
	"context"
	"os"
	sdkgo "github.com/apideck-libraries/sdk-go"
	"github.com/apideck-libraries/sdk-go/models/operations"
	"log"
)

func main() {
    ctx := context.Background()
    
    s := sdkgo.New(
        sdkgo.WithSecurity(os.Getenv("APIDECK_API_KEY")),
        sdkgo.WithConsumerID("test-consumer"),
        sdkgo.WithAppID("dSBdXd2H6Mqwfg0atXHXYcysLJE9qyn1VwBtXHX"),
    )

    res, err := s.Vault.CustomFields.List(ctx, operations.VaultCustomFieldsAllRequest{
        UnifiedAPI: "crm",
        ServiceID: "pipedrive",
        Resource: "leads",
        ResourceID: sdkgo.String("1234"),
    })
    if err != nil {
        log.Fatal(err)
    }
    if res.GetCustomFieldsResponse != nil {
        // handle response
    }
}
```
